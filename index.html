    <script>

        const desarrollos = [

            { cat: 'Core', desc: 'Capitas (MultiSponsor - WMMS)' },

            { cat: 'Core', desc: 'Reembolsos (Independiente/MultiDist)' },

            { cat: 'GOLS', desc: 'Multimoneda' },

            { cat: 'GOLS', desc: 'Multiidioma' },

            { cat: 'GOLS', desc: 'Campañas Promocionales' },

            { cat: 'GOLS', desc: 'Carrito Embebible' },

            { cat: 'SelfService', desc: 'Imágenes (Voucher/Mails/Interfaz)' },

            { cat: 'SelfService', desc: 'Modelo Vouchers (Drag & Drop)' },

            { cat: 'SelfService', desc: 'Textos Beneficios (Sync Audax)' },

            { cat: 'SelfService', desc: 'Editor de Mails (Externo)' },

            { cat: 'Módulos', desc: 'Comisiones (Canal/Ventas)' },

            { cat: 'Módulos', desc: 'Precompras' },

            { cat: 'Módulos', desc: 'Liquidaciones Facturación' },

            { cat: 'Módulos', desc: 'Dashboard para Distribuidor' },

            { cat: 'Integración', desc: 'ERP Standard' },

            { cat: 'Integración', desc: 'Pasarela de Pagos (Extra Stripe)' },

            { cat: 'Integración', desc: 'Emisores Externos (Simply POS)' }

        ];

 

        function renderTabla() {

            const body = document.getElementById('tabla-body');

            body.innerHTML = desarrollos.map((d, index) => `

                <tr class="hover:bg-blue-50 transition border-b border-gray-100 row-item">

                    <td class="p-4 font-semibold text-slate-500">${d.cat}</td>

                    <td class="p-4 font-medium text-slate-900">${d.desc}</td>

                    <td class="p-4 text-center"><input type="number" min="1" max="5" value="0" oninput="updateRow(this)" class="w-12 text-center border rounded p-1 score-input"></td>

                    <td class="p-4 text-center"><input type="number" min="1" max="5" value="0" oninput="updateRow(this)" class="w-12 text-center border rounded p-1 score-input"></td>

                    <td class="p-4 text-center"><input type="number" min="1" max="5" value="0" oninput="updateRow(this)" class="w-12 text-center border rounded p-1 score-input"></td>

                    <td class="p-4 text-center"><input type="number" min="1" max="5" value="0" oninput="updateRow(this)" class="w-12 text-center border rounded p-1 score-input"></td>

                    <td class="p-4 text-center bg-slate-50 font-bold total-cell">0</td>

                    <td class="p-4 text-center font-bold text-blue-600 rank-cell">-</td>

                </tr>

            `).join('');

        }

 

        function updateRow(input) {

            if (input.value > 5) input.value = 5;

            if (input.value < 0) input.value = 0;

           

            calculateAll();

        }

 

        function calculateAll() {

            const rows = document.querySelectorAll('.row-item');

            const totals = [];

 

            rows.forEach(row => {

                const inputs = row.querySelectorAll('.score-input');

                let sum = 0;

                inputs.forEach(i => sum += parseInt(i.value || 0));

               

                const totalCell = row.querySelector('.total-cell');

                totalCell.innerText = sum;

                totals.push({ row, sum });

            });

 

            // Ordenar para ranking (solo si hay valores)

            const sorted = [...totals].sort((a, b) => b.sum - a.sum);

            totals.forEach(item => {

                const rank = sorted.findIndex(s => s.sum === item.sum) + 1;

                item.row.querySelector('.rank-cell').innerText = item.sum > 0 ? rank : '-';

            });

        }

 

        function exportToCSV() {

            let csv = 'Categoria;Desarrollo;Dificultad;Monetizacion;Venta;Valor;Total;Ranking\n';

            const rows = document.querySelectorAll('.row-item');

           

            rows.forEach(row => {

                const cells = row.querySelectorAll('td');

                const inputs = row.querySelectorAll('.score-input');

                const cat = cells[0].innerText;

                const desc = cells[1].innerText;

                const total = row.querySelector('.total-cell').innerText;

                const rank = row.querySelector('.rank-cell').innerText;

               

                csv += `${cat};${desc};${inputs[0].value};${inputs[1].value};${inputs[2].value};${inputs[3].value};${total};${rank}\n`;

            });

 

            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            const url = URL.createObjectURL(blob);

            link.setAttribute("href", url);

            link.setAttribute("download", "prioridades_garlan_2026.csv");

            link.style.visibility = 'hidden';

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

        }

 

        window.onload = renderTabla;
